## 一、目标

实现平台级管理能力，包括租户/组织管理、订阅与计费、审计日志、系统监控，为 SaaS 运营提供基础设施。

## 三、订阅与计费

### 3.1 数据表

```sql
CREATE TABLE subscriptions (
    id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    org_node_id   BIGINT UNSIGNED NOT NULL,                -- 关联到顶层组织节点
    plan          VARCHAR(32)     NOT NULL DEFAULT 'free',  -- free / standard / premium
    status        VARCHAR(16)     NOT NULL DEFAULT 'active',-- active / suspended / expired
    max_employees INT             NOT NULL DEFAULT 20,      -- 最大员工数
    max_ai_tokens INT             NOT NULL DEFAULT 10000,   -- 月 AI token 配额
    start_date    DATE            NOT NULL,
    end_date      DATE,                                     -- NULL = 永不过期
    created_at    DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at    DATETIME        NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    UNIQUE KEY uk_org (org_node_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 套餐定义
-- free:     20 员工, 10K AI tokens/月, 基础功能
-- standard: 200 员工, 100K AI tokens/月, 全部功能
-- premium:  不限员工, 500K AI tokens/月, 优先支持
```

### 3.2 配额检查

```go
// internal/platform/subscription/checker.go

type Checker struct {
    repo SubscriptionRepository
}

func (c *Checker) CheckEmployeeQuota(ctx context.Context, orgNodeID uint64) error {
    sub, err := c.repo.GetByOrgNode(ctx, orgNodeID)
    if err != nil { return err }
    
    currentCount, err := c.repo.CountEmployees(ctx, orgNodeID)
    if err != nil { return err }
    
    if currentCount >= sub.MaxEmployees {
        return errors.SubscriptionQuotaExceeded("employee", currentCount, sub.MaxEmployees)
    }
    return nil
}
```

### 4.1 数据表

```sql
CREATE TABLE audit_logs (
    id            BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    org_node_id   BIGINT UNSIGNED,                         -- NULL = 平台级操作
    user_id       BIGINT UNSIGNED NOT NULL,

    INDEX idx_user (user_id, created_at),
    INDEX idx_action (action, created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 4.2 审计中间件

```go
// internal/platform/audit/middleware.go

func AuditMiddleware(logger *AuditLogger) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            // 仅记录写操作
            if r.Method == "GET" || r.Method == "HEAD" || r.Method == "OPTIONS" {
                next.ServeHTTP(w, r)
                return
            }
            
            // 包装 ResponseWriter 以捕获状态码
            rw := &responseWrapper{ResponseWriter: w}
            next.ServeHTTP(rw, r)
            
            // 异步写审计日志
            if rw.statusCode < 400 {
                go logger.Log(r.Context(), AuditEntry{
                    Action:     deriveAction(r),
                    Resource:   deriveResource(r),
                    StatusCode: rw.statusCode,
                    IP:         r.RemoteAddr,
                    UserAgent:  r.UserAgent(),
                })
            }
        })
    }
}
```

## 五、系统监控

### 5.1 健康检查端点

```go
// GET /healthz
func HealthHandler(db *gorm.DB, rdb *redis.Client) http.HandlerFunc {
    return func(w http.ResponseWriter, r *http.Request) {
        checks := map[string]string{}
        
        // MySQL
        if err := db.Exec("SELECT 1").Error; err != nil {
            checks["mysql"] = "unhealthy: " + err.Error()
        } else {
            checks["mysql"] = "ok"
        }
        
        // Redis
        if err := rdb.Ping(r.Context()).Err(); err != nil {
            checks["redis"] = "unhealthy: " + err.Error()
        } else {
            checks["redis"] = "ok"
        }
        
        // 整体状态
        healthy := true
        for _, v := range checks {
            if v != "ok" { healthy = false; break }
        }
        
        status := http.StatusOK
        if !healthy { status = http.StatusServiceUnavailable }
        
        w.WriteHeader(status)
        json.NewEncoder(w).Encode(map[string]any{
            "status": map[bool]string{true: "healthy", false: "unhealthy"}[healthy],
            "checks": checks,
        })
    }
}
```
