### 2.2 V3 → V4 数据迁移脚本

```sql
-- 100001_migrate_v3_data.up.sql

-- 1. 创建默认组织节点（V3 为单租户，迁移为默认组织）
INSERT INTO org_nodes (id, name, type, parent_id, path, status)
VALUES (1, '默认组织', 'org', NULL, '/1', 'active');

INSERT INTO org_nodes (id, name, type, parent_id, path, status)
VALUES (2, '默认院区', 'campus', 1, '/1/2', 'active');

-- 2. 迁移员工数据（假设 V3 表名 v3_employees）
INSERT INTO employees (name, employee_no, phone, position, org_node_id, status, created_at, updated_at)
SELECT 
    name, 
    employee_no, 
    phone, 
    COALESCE(position, ''),
    2,           -- 默认归入默认院区
    'active',
    created_at, 
    COALESCE(updated_at, created_at)
FROM v3_employees
WHERE deleted_at IS NULL;

-- 3. 迁移班次数据
INSERT INTO shifts (name, code, color, start_time, end_time, org_node_id, created_at, updated_at)
SELECT 
    name, 
    code, 
    COALESCE(color, '#409EFF'),
    start_time, 
    end_time,
    2,
    created_at, 
    COALESCE(updated_at, created_at)
FROM v3_shifts
WHERE deleted_at IS NULL;

-- 4. 迁移用户账号
INSERT INTO users (username, password_hash, display_name, created_at, updated_at)
SELECT 
    username, 
    password,   -- 假设 V3 已使用 bcrypt
    COALESCE(nickname, username),
    created_at, 
    COALESCE(updated_at, created_at)
FROM v3_users
WHERE deleted_at IS NULL;

-- 5. 为迁移用户分配默认角色
INSERT INTO user_node_roles (user_id, org_node_id, role)
SELECT u.id, 2, 'scheduler'
FROM users u
INNER JOIN v3_users v ON u.username = v.username;

-- 6. 清理 V3 表（可选，建议保留备份）
-- RENAME TABLE v3_employees TO _backup_v3_employees;
-- RENAME TABLE v3_shifts TO _backup_v3_shifts;
-- RENAME TABLE v3_users TO _backup_v3_users;
```

### 2.3 数据验证脚本

```sql
-- verify_migration.sql

-- 检查员工数量一致
SELECT 'employees' AS entity,
    (SELECT COUNT(*) FROM v3_employees WHERE deleted_at IS NULL) AS v3_count,
    (SELECT COUNT(*) FROM employees) AS v4_count;

-- 检查班次数量一致
SELECT 'shifts' AS entity,
    (SELECT COUNT(*) FROM v3_shifts WHERE deleted_at IS NULL) AS v3_count,
    (SELECT COUNT(*) FROM shifts) AS v4_count;

-- 检查用户数量一致
SELECT 'users' AS entity,
    (SELECT COUNT(*) FROM v3_users WHERE deleted_at IS NULL) AS v3_count,
    (SELECT COUNT(*) FROM users) AS v4_count;

-- 检查所有员工都有 org_node_id
SELECT COUNT(*) AS orphan_employees FROM employees WHERE org_node_id IS NULL OR org_node_id = 0;

-- 检查组织树完整性
SELECT COUNT(*) AS invalid_paths FROM org_nodes WHERE path = '' OR path IS NULL;
```

## 三、Go 迁移工具集成

```go
// cmd/migrate/main.go
package main

import (
    "flag"
    "log"
    
    "github.com/golang-migrate/migrate/v4"
    _ "github.com/golang-migrate/migrate/v4/database/mysql"
    _ "github.com/golang-migrate/migrate/v4/source/file"
)

func main() {
    dsn := flag.String("dsn", "", "MySQL DSN")
    dir := flag.String("dir", "migrations", "Migrations directory")
    action := flag.String("action", "up", "up / down / version / force")
    steps := flag.Int("steps", 0, "Number of steps (0 = all)")
    flag.Parse()

    m, err := migrate.New("file://"+*dir, "mysql://"+*dsn)
    if err != nil { log.Fatal(err) }

    switch *action {
    case "up":
        if *steps > 0 {
            err = m.Steps(*steps)
        } else {
            err = m.Up()
        }
    case "down":
        if *steps > 0 {
            err = m.Steps(-*steps)
        } else {
            err = m.Down()
        }
    case "version":
        v, dirty, _ := m.Version()
        log.Printf("Version: %d, Dirty: %v", v, dirty)
        return
    case "force":
        v := uint(*steps)
        err = m.Force(int(v))
    }

    if err != nil && err != migrate.ErrNoChange {
        log.Fatal(err)
    }
    log.Println("Migration completed successfully")
}
```

## 四、Docker 部署

### 4.1 Dockerfile（多阶段构建）

```dockerfile
# ===== Stage 1: 前端构建 =====
FROM node:20-alpine AS frontend-builder
WORKDIR /app/frontend
COPY frontend/web/package.json frontend/web/pnpm-lock.yaml ./
RUN corepack enable && pnpm install --frozen-lockfile
COPY frontend/web/ ./
RUN pnpm build

# ===== Stage 2: Go 构建 =====
FROM golang:1.23-alpine AS backend-builder
RUN apk add --no-cache git
WORKDIR /app
COPY go.mod go.sum ./
COPY pkg/go.mod pkg/go.sum ./pkg/
RUN go mod download
COPY . .
COPY --from=frontend-builder /app/frontend/web/dist ./frontend/web/dist
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w -X main.version=$(git describe --tags --always)" \
  -o /bin/management-service ./cmd/services/management-service
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w -X main.version=$(git describe --tags --always)" \
  -o /bin/rostering-server ./cmd/mcp-servers/rostering-server
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w -X main.version=$(git describe --tags --always)" \
  -o /bin/context-server ./cmd/mcp-servers/context-server
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w -X main.version=$(git describe --tags --always)" \
  -o /bin/rostering-agent ./cmd/agents/rostering

# ===== Stage 3: 运行镜像 =====
FROM alpine:3.20
RUN apk add --no-cache ca-certificates tzdata
ENV TZ=Asia/Shanghai
WORKDIR /app
COPY --from=backend-builder /bin/management-service /app/bin/management-service
COPY --from=backend-builder /bin/rostering-server /app/bin/rostering-server
COPY --from=backend-builder /bin/context-server /app/bin/context-server
COPY --from=backend-builder /bin/rostering-agent /app/bin/rostering-agent
COPY config/ /app/config/
COPY migrations/ /app/migrations/
EXPOSE 8080 8081 8082 8083
```

### 6.2 上线步骤

```
1. 备份 V3 数据库
   $ mysqldump -u root -p gantt_v3 > backup_v3_$(date +%Y%m%d).sql

2. 创建 V4 数据库
   $ mysql -u root -p -e "CREATE DATABASE gantt_v4 CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci"

3. 执行结构迁移
   $ ./migrate -dsn "gantt:pass@tcp(localhost:3306)/gantt_v4" -action up

4. 执行数据迁移
   $ mysql -u root -p gantt_v4 < migrations/100001_migrate_v3_data.up.sql

5. 验证数据
   $ mysql -u root -p gantt_v4 < scripts/verify_migration.sql

6. 部署新版本
   $ docker compose up -d --build
   # 或
   $ kubectl apply -f k8s/

7. 验证健康检查
   $ curl http://localhost:8080/healthz

8. 切换 DNS/Ingress 指向新服务

9. 观察监控 30 分钟，确认无异常
```

### 6.3 回滚方案

```
# 如需回滚：
1. 切换 DNS/Ingress 回 V3
2. 停止 V4 服务
   $ docker compose down
3. V3 数据库已有备份，无需额外操作
4. 分析问题后修复，重新执行上线流程
```

## 七、CI/CD 流水线建议

```yaml
# .github/workflows/ci.yml (示例)
name: CI/CD
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest
    services:
      mysql:
        image: mysql:8.0
        env:
          MYSQL_ROOT_PASSWORD: test
          MYSQL_DATABASE: gantt_test
        ports: [3306:3306]
      redis:
        image: redis:7-alpine
        ports: [6379:6379]
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-go@v5
        with: { go-version: '1.23' }
      - run: go test ./... -race -coverprofile=coverage.out
      - uses: actions/setup-node@v4
        with: { node-version: '20' }
      - run: cd frontend/web && corepack enable && pnpm install && pnpm lint && pnpm build

  deploy:
    needs: test
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - run: docker build -t registry.example.com/gantt-server:${{ github.sha }} .
      - run: docker push registry.example.com/gantt-server:${{ github.sha }}
```
