## 二、数据库设计

### 2.1 schedules

```sql
CREATE TABLE schedules (
    id            VARCHAR(64) PRIMARY KEY,
    org_node_id   VARCHAR(64) NOT NULL,
    name          VARCHAR(128) NOT NULL,
    start_date    DATE NOT NULL,
    end_date      DATE NOT NULL,
    status        ENUM('draft', 'generating', 'review', 'published', 'archived') NOT NULL DEFAULT 'draft',
    pipeline_type VARCHAR(32) NOT NULL DEFAULT 'deterministic',  -- deterministic / ai_assisted
    config        JSON DEFAULT NULL,                              -- 排班配置（参与班次、人数需求等）
    created_by    VARCHAR(64) NOT NULL,
    created_at    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_org_node (org_node_id),
    INDEX idx_status (org_node_id, status),
    INDEX idx_date (org_node_id, start_date, end_date)
);
```

### 2.2 schedule_assignments

```sql
CREATE TABLE schedule_assignments (
    id            VARCHAR(64) PRIMARY KEY,
    schedule_id   VARCHAR(64) NOT NULL,
    org_node_id   VARCHAR(64) NOT NULL,
    employee_id   VARCHAR(64) NOT NULL,
    shift_id      VARCHAR(64) NOT NULL,
    date          DATE NOT NULL,
    source        ENUM('fixed', 'rule', 'fill', 'ai', 'manual') NOT NULL,  -- 排班来源
    created_at    DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (schedule_id) REFERENCES schedules(id),
    FOREIGN KEY (employee_id) REFERENCES employees(id),
    FOREIGN KEY (shift_id) REFERENCES shifts(id),
    INDEX idx_org_node (org_node_id),
    INDEX idx_schedule (schedule_id),
    INDEX idx_emp_date (employee_id, date),
    UNIQUE KEY uk_assignment (schedule_id, employee_id, shift_id, date)
);
```

### 3.2 Step 接口

```go
// step/step.go

type Step interface {
    Name() string
    Execute(ctx context.Context, state *ScheduleState) error
}

// ScheduleState — Pipeline 中流转的共享状态
type ScheduleState struct {
    // 输入
    Schedule      *Schedule
    Config        *ScheduleConfig       // 参与班次、人数需求
    EffectiveRules []rule.Rule          // 生效规则集

    // 中间状态
    ShiftOrder    []shift.Shift         // 拓扑排序后的班次执行顺序
    Candidates    map[string][]string   // shiftID → 候选员工 ID 列表
    Scores        map[string]int        // employeeID → 偏好评分

    // 输出
    Assignments   []Assignment          // 排班结果
    Violations    []Violation           // 规则违反记录
    
    // 进度回调（用于 WebSocket 推送）
    OnProgress    func(step string, progress float64, message string)
}

type ScheduleConfig struct {
    ShiftIDs      []string                          // 参与排班的班次
    Requirements  map[string]map[string]int          // shiftID → date → 需求人数
    Preferences   []PersonalPreference               // 个人偏好
}
```

### 3.3 Pipeline 执行器

```go
// pipeline/pipeline.go

type Pipeline struct {
    name  string
    steps []step.Step
}

func NewPipeline(name string, steps ...step.Step) *Pipeline {
    return &Pipeline{name: name, steps: steps}
}

func (p *Pipeline) Run(ctx context.Context, state *step.ScheduleState) error {
    logger.Info("pipeline started", zap.String("pipeline", p.name))

    for i, s := range p.steps {
        logger.Info("step started",
            zap.String("pipeline", p.name),
            zap.String("step", s.Name()),
            zap.Int("index", i+1),
            zap.Int("total", len(p.steps)),
        )

        // 进度回调
        if state.OnProgress != nil {
            progress := float64(i) / float64(len(p.steps))
            state.OnProgress(s.Name(), progress, "执行中")
        }

        if err := s.Execute(ctx, state); err != nil {
            logger.Error("step failed",
                zap.String("step", s.Name()),
                zap.Error(err),
            )
            return fmt.Errorf("step '%s' failed: %w", s.Name(), err)
        }
    }

    if state.OnProgress != nil {
        state.OnProgress("completed", 1.0, "排班完成")
    }

    logger.Info("pipeline completed", zap.String("pipeline", p.name))
    return nil
}
```

```go
// step/load_rules.go

type LoadRulesStep struct {
    ruleService *rule.RuleService
}

func (s *LoadRulesStep) Name() string { return "LoadRules" }

func (s *LoadRulesStep) Execute(ctx context.Context, state *ScheduleState) error {
    nodeID := tenant.GetOrgNodeID(ctx)
    effectiveRules, err := s.ruleService.ComputeEffectiveRules(ctx, nodeID)
    if err != nil {
        return err
    }
    state.EffectiveRules = effectiveRules.Rules

    // 基于依赖关系进行拓扑排序
    state.ShiftOrder, err = s.shiftService.GetTopologicalOrder(ctx)
    return err
}
```

#### FilterCandidatesStep

```go
// step/filter_candidates.go

type FilterCandidatesStep struct {
    empRepo   *employee.Repository
    leaveRepo *leave.Repository
}

func (s *FilterCandidatesStep) Name() string { return "FilterCandidates" }

func (s *FilterCandidatesStep) Execute(ctx context.Context, state *ScheduleState) error {
    // 1. 查询所有 active 员工
    employees, _ := s.empRepo.ListActive(ctx)

    // 2. 查询排班周期内的请假记录
    leaves, _ := s.leaveRepo.ListByDateRange(ctx, state.Schedule.StartDate, state.Schedule.EndDate)
    leaveMap := buildLeaveMap(leaves) // employeeID+date → true

    // 3. 为每个班次每天过滤候选人
    for _, shift := range state.ShiftOrder {
        for date := state.Schedule.StartDate; !date.After(state.Schedule.EndDate); date = date.AddDate(0, 0, 1) {
            candidates := []string{}
            for _, emp := range employees {
                // 排除请假
                if leaveMap[emp.ID+date.Format("2006-01-02")] { continue }
                // 排除已占位
                if isOccupied(state.Assignments, emp.ID, date) { continue }
                candidates = append(candidates, emp.ID)
            }
            key := shift.ID + "|" + date.Format("2006-01-02")
            state.Candidates[key] = candidates
        }
    }
    return nil
}
```

#### PhaseTwoStep（兜底填充）

```go
func (s *PhaseTwoStep) Execute(ctx context.Context, state *ScheduleState) error {
    for _, shift := range state.ShiftOrder {
        for date := range dateRange {
            needed := state.Config.Requirements[shift.ID][dateStr] - assigned
            if needed <= 0 { continue }

            candidates := state.Candidates[key]
            // 按偏好评分排序
            scored := scoreCandidates(candidates, state.EffectiveRules, shift.ID, date)
            sort.Slice(scored, func(i, j int) bool { return scored[i].Score > scored[j].Score })

            // 取前 N 个，同分随机
            selected := selectTopN(scored, needed)
            for _, emp := range selected {
                state.Assignments = append(state.Assignments, Assignment{
                    EmployeeID: emp.ID, ShiftID: shift.ID, Date: date, Source: "fill",
                })
            }
        }
    }
}
```

#### FullValidationStep

```go
func (s *FullValidationStep) Execute(ctx context.Context, state *ScheduleState) error {
    // 对所有 Assignments 执行全规则扫描
    for _, assignment := range state.Assignments {
        results := checker.ValidateAll(state.EffectiveRules, &checker.CheckContext{
            EmployeeID:  assignment.EmployeeID,
            ShiftID:     assignment.ShiftID,
            Date:        assignment.Date,
            Assignments: state.Assignments,
        })
        for _, r := range results {
            if !r.Pass {
                state.Violations = append(state.Violations, Violation{
                    AssignmentID: assignment.ID,
                    Rule:         r.RuleName,
                    Reason:       r.Reason,
                })
            }
        }
    }
    return nil  // 违规记录在 state.Violations 中，不中断 Pipeline
}
```

## 六、验收标准

| 验收项 | 通过条件 |
|--------|---------|
| 确定性排班 | 50 人 × 3 班次 × 7 天，排班结果完整无缺漏 |
| 规则执行 | 排他班次不出现在同一人同一天 |
| 兜底填充 | 所有需求人数均被满足（或报告不可满足原因） |
| 全规则校验 | 校验结果包含所有违规记录 |
| 排班调整 | 手动修改后自动校验，报告新增违规 |
| 性能 | 100 规则 × 50 人 × 7 天 < 500ms |
| WebSocket | 排班过程中前端实时收到进度更新 |
| 变更记录 | 每次手动调整都有完整的变更日志 |
