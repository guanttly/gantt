## 四、核心实现

### 4.1 Axios 实例与 Token 刷新

```typescript
// src/api/client.ts
import axios from 'axios'
import { useAuthStore } from '@/stores/auth'

const client = axios.create({
  baseURL: '/api/v1',
  timeout: 15000,
})

// 请求拦截：自动附加 token
client.interceptors.request.use((config) => {
  const auth = useAuthStore()
  if (auth.accessToken) {
    config.headers.Authorization = `Bearer ${auth.accessToken}`
  }
  return config
})

// 响应拦截：401 自动刷新 token
let isRefreshing = false
let pendingRequests: Array<(token: string) => void> = []

client.interceptors.response.use(
  (response) => response,
  async (error) => {
    const originalRequest = error.config
    const auth = useAuthStore()
    
    if (error.response?.status === 401 && !originalRequest._retry) {
      if (isRefreshing) {
        return new Promise((resolve) => {
          pendingRequests.push((token) => {
            originalRequest.headers.Authorization = `Bearer ${token}`
            resolve(client(originalRequest))
          })
        })
      }
      
      originalRequest._retry = true
      isRefreshing = true
      
      try {
        const newToken = await auth.refreshToken()
        pendingRequests.forEach((cb) => cb(newToken))
        pendingRequests = []
        originalRequest.headers.Authorization = `Bearer ${newToken}`
        return client(originalRequest)
      } catch {
        auth.logout()
        window.location.href = '/login'
      } finally {
        isRefreshing = false
      }
    }
    return Promise.reject(error)
  }
)

export default client
```

### 4.2 路由守卫

```typescript
// src/router/guards.ts
import type { Router } from 'vue-router'
import { useAuthStore } from '@/stores/auth'

const PUBLIC_ROUTES = ['/login']

export function setupGuards(router: Router) {
  router.beforeEach(async (to, from, next) => {
    const auth = useAuthStore()
    
    // 公开页面
    if (PUBLIC_ROUTES.includes(to.path)) {
      if (auth.isLoggedIn) return next('/') // 已登录跳首页
      return next()
    }
    
    // 未登录
    if (!auth.isLoggedIn) {
      return next({ path: '/login', query: { redirect: to.fullPath } })
    }
    
    // 未选择节点
    if (!auth.currentNodeId && to.path !== '/select-node') {
      return next('/select-node')
    }
    
    // 权限检查
    if (to.meta.requiredRole) {
      const requiredRole = to.meta.requiredRole as string
      if (!auth.hasRole(requiredRole)) {
        return next('/403')
      }
    }
    
    next()
  })
}
```

### 4.4 认证 Store

```typescript
// src/stores/auth.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { login as apiLogin, refreshToken as apiRefresh, switchNode as apiSwitch } from '@/api/auth'

export const useAuthStore = defineStore('auth', () => {
  const accessToken = ref<string | null>(localStorage.getItem('access_token'))
  const refreshTokenValue = ref<string | null>(localStorage.getItem('refresh_token'))
  const user = ref<User | null>(null)
  const currentNodeId = ref<number | null>(null)
  const currentNodePath = ref<string>('')
  const currentRole = ref<string>('')
  const availableNodes = ref<OrgNode[]>([])

  const isLoggedIn = computed(() => !!accessToken.value)

  function hasRole(role: string): boolean {
    const hierarchy = ['employee', 'scheduler', 'dept_admin', 'org_admin', 'platform_admin']
    return hierarchy.indexOf(currentRole.value) >= hierarchy.indexOf(role)
  }

  async function login(username: string, password: string) {
    const res = await apiLogin({ username, password })
    accessToken.value = res.access_token
    refreshTokenValue.value = res.refresh_token
    user.value = res.user
    availableNodes.value = res.available_nodes
    localStorage.setItem('access_token', res.access_token)
    localStorage.setItem('refresh_token', res.refresh_token)
  }

  async function selectNode(nodeId: number) {
    const res = await apiSwitch({ node_id: nodeId })
    accessToken.value = res.access_token
    currentNodeId.value = nodeId
    currentNodePath.value = res.node_path
    currentRole.value = res.role
    localStorage.setItem('access_token', res.access_token)
  }

  async function refreshToken(): Promise<string> {
    const res = await apiRefresh({ refresh_token: refreshTokenValue.value! })
    accessToken.value = res.access_token
    localStorage.setItem('access_token', res.access_token)
    return res.access_token
  }

  function logout() {
    accessToken.value = null
    refreshTokenValue.value = null
    user.value = null
    currentNodeId.value = null
    localStorage.removeItem('access_token')
    localStorage.removeItem('refresh_token')
  }

  return {
    accessToken, user, currentNodeId, currentNodePath, currentRole,
    availableNodes, isLoggedIn,
    hasRole, login, selectNode, refreshToken, logout,
  }
})
```

### 4.5 WebSocket 可组合函数

```typescript
// src/composables/useWebSocket.ts
import { ref, onMounted, onUnmounted } from 'vue'
import { useAuthStore } from '@/stores/auth'

export function useWebSocket() {
  const ws = ref<WebSocket | null>(null)
  const connected = ref(false)
  const messages = ref<WSMessage[]>([])
  
  function connect() {
    const auth = useAuthStore()
    const protocol = location.protocol === 'https:' ? 'wss:' : 'ws:'
    ws.value = new WebSocket(`${protocol}//${location.host}/api/v1/ws?token=${auth.accessToken}`)
    
    ws.value.onopen = () => { connected.value = true }
    ws.value.onclose = () => {
      connected.value = false
      setTimeout(connect, 3000)  // 自动重连
    }
    ws.value.onmessage = (event) => {
      const msg = JSON.parse(event.data)
      messages.value.push(msg)
    }
  }
  
  onMounted(connect)
  onUnmounted(() => ws.value?.close())
  
  return { connected, messages }
}
```
