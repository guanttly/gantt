# M05 — 规则引擎

> **所属阶段**: 阶段二（第 5~6 周）  
> **前置依赖**: M04（基础数据管理）  
> **交付物**: 规则 CRUD + 规则继承合并 + 约束检查器 + 生效规则计算

## 一、目标

实现确定性的规则引擎，支持规则沿组织树继承与覆盖，100% 代码化执行，零 LLM 依赖。排班引擎的各 Step 通过调用规则引擎来完成约束检查和偏好评分。

## 二、数据库设计

### 2.1 rules

```sql
CREATE TABLE rules (
    id              VARCHAR(64) PRIMARY KEY,
    org_node_id     VARCHAR(64) NOT NULL,
    name            VARCHAR(128) NOT NULL,
    category        ENUM('constraint', 'preference', 'dependency') NOT NULL,
    sub_type        VARCHAR(32) NOT NULL,            -- forbid/limit/must/prefer/combinable/source/order
    config          JSON NOT NULL,                    -- 规则参数（结构化）
    priority        INT NOT NULL DEFAULT 0,           -- 同类规则优先级
    is_enabled      BOOLEAN NOT NULL DEFAULT TRUE,
    override_rule_id VARCHAR(64) DEFAULT NULL,        -- 覆盖的上级规则 ID（NULL=新增）
    description     VARCHAR(512) DEFAULT NULL,
    created_at      DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at      DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_org_node (org_node_id),
    INDEX idx_category (org_node_id, category, sub_type),
    INDEX idx_override (override_rule_id)
);
```

### 2.2 rule_associations

```sql
CREATE TABLE rule_associations (
    id            VARCHAR(64) PRIMARY KEY,
    rule_id       VARCHAR(64) NOT NULL,
    org_node_id   VARCHAR(64) NOT NULL,
    target_type   ENUM('shift', 'group', 'employee') NOT NULL,
    target_id     VARCHAR(64) NOT NULL,

    FOREIGN KEY (rule_id) REFERENCES rules(id) ON DELETE CASCADE,
    INDEX idx_org_node (org_node_id),
    INDEX idx_rule (rule_id),
    INDEX idx_target (target_type, target_id)
);
```

func (s *RuleService) ComputeEffectiveRules(ctx context.Context, nodeID string) (*EffectiveRuleSet, error) {
    // 1. 查询当前节点的 path
    node, _ := s.nodeRepo.GetByID(nodeID)

    // 2. 提取祖先节点 ID 列表（按 path 分割，从根到叶）
    ancestorIDs := extractAncestorIDs(node.Path)  // [org1, campusA, deptX]

    // 3. 按层级从上到下查询每个节点的规则
    allRules := []Rule{}
    for _, aid := range ancestorIDs {
        rules, _ := s.repo.ListByNodeID(aid)
        allRules = append(allRules, rules...)
    }

    // 4. 合并：同 category+sub_type+关联目标 的规则，下级覆盖上级
    merged := mergeRules(allRules)

### 3.3 约束检查器

```go
// checker/checker.go

type CheckContext struct {
    EmployeeID   string
    ShiftID      string
    Date         time.Time
    Assignments  []Assignment  // 已有排班
    Candidates   []string      // 候选人列表
}

type CheckResult struct {
    Pass    bool
    Reason  string  // 不通过的原因
}

type Checker interface {
    Type() string                                    // 匹配 rule.sub_type
    Check(ctx context.Context, rule Rule, checkCtx *CheckContext) CheckResult
}

// 检查器注册表
var registry = map[string]Checker{}

func Register(c Checker) { registry[c.Type()] = c }
func Get(subType string) Checker { return registry[subType] }

// 批量校验
func ValidateAll(rules []Rule, checkCtx *CheckContext) []CheckResult {
    results := []CheckResult{}
    for _, rule := range rules {
        if checker := Get(rule.SubType); checker != nil {
            result := checker.Check(context.Background(), rule, checkCtx)
            results = append(results, result)
        }
    }
    return results
}
```

### 3.4 偏好评分

```go
// checker/preference.go

type PreferenceScorer struct{}

func (s *PreferenceScorer) Score(rules []Rule, employeeID string, shiftID string, date time.Time) int {
    score := 0
    for _, rule := range rules {
        if rule.Category != "preference" { continue }
        cfg := parsePreferConfig(rule.Config)
        if cfg.EmployeeID == employeeID && cfg.ShiftID == shiftID {
            score += cfg.Weight
        }
    }
    return score
}
```
