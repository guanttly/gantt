> **所属阶段**: 阶段一（第 1~2 周）

## 一、目标

实现树状组织结构（机构→院区→科室），所有业务数据自动隔离到组织节点级别，杜绝数据越权访问。

## 二、数据库设计

### 2.1 org_nodes 表

```sql
CREATE TABLE org_nodes (
    id          VARCHAR(64) PRIMARY KEY,
    parent_id   VARCHAR(64) DEFAULT NULL,
    node_type   ENUM('organization', 'campus', 'department', 'custom') NOT NULL,
    name        VARCHAR(128) NOT NULL,
    code        VARCHAR(64) NOT NULL,
    path        VARCHAR(512) NOT NULL,              -- 物化路径: /org1/campusA/deptX
    depth       INT NOT NULL DEFAULT 0,
    is_login_point BOOLEAN NOT NULL DEFAULT FALSE,
    status      ENUM('active', 'suspended') NOT NULL DEFAULT 'active',
    created_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at  DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (parent_id) REFERENCES org_nodes(id),
    UNIQUE KEY uk_parent_code (parent_id, code),    -- 同级 code 唯一
    INDEX idx_path (path),                           -- 物化路径查询
    INDEX idx_parent (parent_id),
    INDEX idx_type_status (node_type, status)
);
```

### 2.2 物化路径说明

- 创建节点时自动计算: `path = parent.path + "/" + node.id`
- 查询某节点的所有后代: `WHERE path LIKE '/org1/campusA/%'`
- 查询某节点的所有祖先: 按 `/` 分割 path，依次查找
- 移动节点时需更新该节点及所有后代的 path（批量 UPDATE）

### 3.3 GORM 全局 Scope

```go
// scope.go

// NodeScope — 仅查本节点数据
func NodeScope(nodeID string) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        return db.Where("org_node_id = ?", nodeID)
    }
}

// NodeTreeScope — 查本节点 + 所有后代节点数据
func NodeTreeScope(nodeIDs []string) func(db *gorm.DB) *gorm.DB {
    return func(db *gorm.DB) *gorm.DB {
        return db.Where("org_node_id IN ?", nodeIDs)
    }
}

// GetDescendantNodeIDs — 预查询后代节点 ID 列表（避免子查询性能问题）
func GetDescendantNodeIDs(db *gorm.DB, nodePath string) ([]string, error) {
    var ids []string
    err := db.Model(&OrgNode{}).
        Where("path LIKE ?", nodePath+"%").
        Pluck("id", &ids).Error
    return ids, err
}
```

### 3.4 防漏 Callback

```go
// scope.go

func RegisterTenantGuard(db *gorm.DB) {
    // 注册表名白名单（平台级表不需要 org_node_id）
    platformTables := map[string]bool{
        "org_nodes": true, "users": true, "user_node_roles": true,
    }

    db.Callback().Query().Before("gorm:query").Register("tenant:guard", func(db *gorm.DB) {
        table := db.Statement.Table
        if platformTables[table] {
            return
        }
        // 检查 WHERE 条件是否包含 org_node_id
        if !containsOrgNodeCondition(db.Statement) {
            db.AddError(fmt.Errorf("TENANT GUARD: query on table '%s' missing org_node_id condition", table))
            // 同时发送告警
            logger.Error("tenant isolation violation", zap.String("table", table))
        }
    })
}
```

### 3.5 组织节点中间件

```go
// middleware.go

func OrgNodeMiddleware() func(next http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            // 从 JWT Claims 中提取当前登录的 org_node_id
            claims := auth.GetClaims(r.Context())
            if claims == nil {
                http.Error(w, "unauthorized", 401)
                return
            }

            // 存入 Context
            ctx := context.WithValue(r.Context(), CtxKeyOrgNodeID, claims.OrgNodeID)
            ctx = context.WithValue(ctx, CtxKeyOrgNodePath, claims.OrgNodePath)

            // 检查 ?scope=tree 参数
            if r.URL.Query().Get("scope") == "tree" {
                ctx = context.WithValue(ctx, CtxKeyScopeTree, true)
            }

            next.ServeHTTP(w, r.WithContext(ctx))
        })
    }
}
```

### 3.6 Context 工具

```go
// context.go

func GetOrgNodeID(ctx context.Context) string { ... }
func GetOrgNodePath(ctx context.Context) string { ... }
func IsScopeTree(ctx context.Context) bool { ... }

// ApplyScope — 根据 Context 自动应用 GORM Scope
func ApplyScope(ctx context.Context, db *gorm.DB) *gorm.DB {
    nodeID := GetOrgNodeID(ctx)
    if IsScopeTree(ctx) {
        nodeIDs, _ := GetDescendantNodeIDs(db, GetOrgNodePath(ctx))
        return db.Scopes(NodeTreeScope(nodeIDs))
    }
    return db.Scopes(NodeScope(nodeID))
}
```

## 六、验收标准

| 验收项 | 通过条件 |
|--------|---------|
| 组织树创建 | 创建 机构→院区→科室 三级树，path 自动计算正确 |
| 数据隔离 | 科室 A 登录后查询员工，只返回科室 A 的数据 |
| 上级查看 | 院区登录 + `?scope=tree` 可看到所有下级科室数据 |
| 防漏检测 | 业务查询未带 `org_node_id` 时返回错误并告警 |
| 停用节点 | 停用院区后，该院区下所有科室状态变为 suspended |
| 平台表白名单 | 查询 `org_nodes`、`users` 表不触发防漏检测 |
